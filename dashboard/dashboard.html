Rule Format Reference
# Block domain:
||example.com^

# Allow exception:
@@||example.com^

# Block specific URL:
||ads.example.com/banner^

# Block by type:
||tracker.com^$script

# Block third-party scripts:
||analytics.com^$script,third-party

# Element hiding (global):
##.ad-container

# Element hiding (site-specific):
example.com##.sidebar-ad

# Allow on specific domain:
||cdn.example.com^$domain=example.com

# Comments:
